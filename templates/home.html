from flask import Flask
from flask_bcrypt import Bcrypt

app = Flask(__name__)
bcrypt = Bcrypt(app)

# Generate secure password hash
password_hash = bcrypt
    .generate_password_hash(
        "SecurePass123!"
    )

# Verify password
is_valid = bcrypt
    .check_password_hash(
        password_hash,
        "SecurePass123!"
    )
print(is_valid)  # → True
